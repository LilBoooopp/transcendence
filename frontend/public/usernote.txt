  <script>
	//register
	const result = document.getElementById('result');

    document.getElementById('register').addEventListener('submit', async (e) => {
      e.preventDefault();
      const res = await fetch('/api/users/register', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({
          email: document.getElementById('email').value,
          username: document.getElementById('username').value,
          password: document.getElementById('password').value,
        })
      });
      const data = await res.json();
      document.getElementById('result').textContent = res.ok
        ? 'Compte créé : ' + data.username
        : 'Erreur : ' + data.message;
    });
	//LOGIN 
	document.getElementById('loginForm').addEventListener('submit', async (e) => {
      e.preventDefault();
	  console.log("enter login");
	  //route pour login...
	  //REST POST '/api/auth/login'
      const res = await fetch('/api/auth/login', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({
          username: document.getElementById('loginUsername').value,
          password: document.getElementById('loginPassword').value,
        })
      });
	  ///verifier si c est bien du json
      const data = await res.json().catch(() => ({ message: 'Erreur serveur'}));
	  if (res.ok) {
    // Changez access_token en accessToken
    localStorage.setItem('authToken', data.accessToken);
    result.textContent = `Connecté : ${data.username || 'OK'}`;
  } else {
    result.textContent = `Erreur : ${data.message || res.status}`;
  }
});
	//LOGOUT
	document.getElementById('logoutBtn').addEventListener('click', (e) => {
  e.preventDefault();
  
  // Supprimer le token du localStorage
  localStorage.removeItem('authToken');
  
  // Effacer le message de connexion
  document.getElementById('result').textContent = 'Vous avez été déconnecté';
  
  // Optionnel: rediriger vers la page de login après 1 seconde
 // setTimeout(() => {
 //   window.location.href = '/'; // ou la page de login
 // }, 1000);
});
	//ENTER GAME
document.getElementById('enterGameBtn').addEventListener('click', async (e) => {
  e.preventDefault();
	console.log("you want to enter the game start");
  // Vérifier si l'utilisateur est connecté (token existe)
  const token = localStorage.getItem('authToken');
  
  if (!token) {
    document.getElementById('result').textContent = 'Erreur: Vous devez être connecté pour accéder au jeu';
    return;
  }
  console.log("you want to enter the game");
  // Rediriger vers la page jeu
  window.location.href = '/game';
});
  </script>

    document.getElementById('showUsersBtn').addEventListener('click', async (e) => {

  if (!token) {

      const res = await fetch('/api/users/all', {

      const users = await res.json();
      if (res.ok) {

    document.getElementById('result').innerHTML = `<h3>Users:</h3>${userList}`;
